### 1. 数据初始化 Hook

#### `src/hooks/useDemoInit.ts`
负责在应用加载时，将我们在 `src/data/` 准备好的探店 Fallback 数据灌入 Zustand Store。

```typescript
import { useEffect } from 'react';
import { useCubeStore } from '../store/useCubeStore';
import { demoScenarios } from '../data/demoScenarios';
import { UiCubeModel } from '../types/cube';

export const useDemoInit = () => {
  const { currentCubeView, setCurrentCubeView } = useCubeStore();

  useEffect(() => {
    // 如果还没加载数据，则默认加载探店决策魔方的初始态
    if (!currentCubeView) {
      const defaultScenario = demoScenarios.find(s => s.id === 'food_decision');
      if (defaultScenario && defaultScenario.heroCube) {
        setCurrentCubeView(defaultScenario.heroCube as UiCubeModel);
      }
    }
  }, [currentCubeView, setCurrentCubeView]);
};
```

### 2. 右侧详情面板

#### `src/components/panels/RightPanel.tsx`
绑定 `useCubeStore` 中的 `selectedFaceId`。当点击 3D 魔方的特定面时，这里会实时响应展示。

```tsx
import React from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { useCubeStore } from '../../store/useCubeStore';
import { CubeFaceId } from '../../types/api';

const accentColors = {
  cyan: 'text-cyan-600 bg-cyan-100/50 border-cyan-200',
  blue: 'text-blue-600 bg-blue-100/50 border-blue-200',
  violet: 'text-violet-600 bg-violet-100/50 border-violet-200',
  amber: 'text-amber-600 bg-amber-100/50 border-amber-200',
  red: 'text-red-600 bg-red-100/50 border-red-200',
  neutral: 'text-slate-600 bg-slate-100/50 border-slate-200',
};

export const RightPanel: React.FC = () => {
  const { currentCubeView, selectedFaceId, setSelectedFaceId } = useCubeStore();

  // 如果没有选中特定的面，默认展示主概览
  const isSelected = !!selectedFaceId;
  const displayFaceId: CubeFaceId = selectedFaceId || 'primary_card';
  const faceData = currentCubeView?.faces?.[displayFaceId];

  return (
    <AnimatePresence>
      <motion.div
        initial={{ x: 100, opacity: 0 }}
        animate={{ x: 0, opacity: 1 }}
        className="absolute right-6 top-24 bottom-24 w-80 z-30 flex flex-col pointer-events-auto"
      >
        <div className="glass-panel h-full rounded-2xl flex flex-col overflow-hidden shadow-xl shadow-slate-200/50 border border-white/60">
          
          {/* Header */}
          <div className="px-5 py-4 border-b border-white/50 bg-white/30 backdrop-blur-md relative">
            {isSelected && (
              <button 
                onClick={() => setSelectedFaceId(null)}
                className="absolute right-4 top-4 text-slate-400 hover:text-slate-700 text-sm"
              >
                ✕ 回到概览
              </button>
            )}
            <h2 className="text-lg font-bold text-slate-800 tracking-wide mt-1">
              {faceData ? faceData.label : '解析概览'}
            </h2>
            <div className="flex items-center gap-2 mt-2">
              <span className={`px-2 py-0.5 text-xs font-medium rounded border ${faceData ? accentColors[faceData.accent] : accentColors.neutral}`}>
                {faceData ? faceData.status.replace('_', ' ').toUpperCase() : 'READY'}
              </span>
              {faceData?.needsConfirmation && (
                <span className="px-2 py-0.5 text-xs font-medium rounded border border-amber-300 text-amber-600 bg-amber-50">
                  待确认
                </span>
              )}
            </div>
          </div>

          {/* Content Body */}
          <div className="flex-1 p-5 overflow-y-auto space-y-4 scrollbar-hide">
            {faceData ? (
              <>
                {/* 模拟的 Summary 块 */}
                <div className="space-y-2">
                  <h3 className="text-sm font-semibold text-slate-500">关键摘要</h3>
                  <div className="glass-pill rounded-xl p-3 flex flex-col gap-2 bg-white/40">
                    {faceData.summaryItems.map((item, idx) => (
                      <div key={idx} className="text-sm text-slate-700 flex items-center gap-2">
                        <span className="w-1.5 h-1.5 rounded-full bg-blue-400" />
                        {item}
                      </div>
                    ))}
                  </div>
                </div>

                {/* 操作与人工补充入口 */}
                <div className="pt-4 border-t border-white/40">
                  <h3 className="text-sm font-semibold text-slate-500 mb-3">相关操作</h3>
                  <div className="flex flex-col gap-2">
                    <button className="glass-button w-full py-2 rounded-lg text-sm font-medium text-blue-600 flex items-center justify-center gap-1">
                      <span>＋</span> 补充 / 修正内容
                    </button>
                    {faceData.id === 'snapshot' && (
                      <button className="bg-gradient-to-r from-blue-500 to-cyan-500 text-white shadow-md shadow-blue-500/20 hover:shadow-blue-500/40 transition-all w-full py-2 rounded-lg text-sm font-medium">
                        保存入柜
                      </button>
                    )}
                  </div>
                </div>
              </>
            ) : (
              <div className="text-sm text-slate-500 flex items-center justify-center h-full">
                加载中...
              </div>
            )}
          </div>
        </div>
      </motion.div>
    </AnimatePresence>
  );
};
```

### 3. 左侧资产陈列柜

#### `src/components/cabinet/CabinetSidebar.tsx`
陈列柜栏，具备“收起”和“半展开”两种状态，读取我们之前写的 fallback 数据库。

```tsx
import React from 'react';
import { motion } from 'framer-motion';
import { useCabinetStore } from '../../store/useCabinetStore';

export const CabinetSidebar: React.FC = () => {
  const { cabinetState, setCabinetState, shelves, assets } = useCabinetStore();

  const isExpanded = cabinetState === 'half' || cabinetState === 'expanded';

  return (
    <motion.div
      initial={false}
      animate={{ width: isExpanded ? 320 : 64 }}
      className="absolute left-0 top-16 bottom-0 z-30 flex pointer-events-auto shadow-[4px_0_24px_rgba(0,0,0,0.02)]"
    >
      {/* 主面板：玻璃陈列柜主体 */}
      <div className="glass-panel border-l-0 rounded-r-2xl h-full w-full flex flex-col overflow-hidden relative border-y-0">
        
        {/* Toggle Button Area */}
        <div 
          className="h-14 border-b border-white/50 flex items-center px-4 cursor-pointer hover:bg-white/30 transition-colors"
          onClick={() => setCabinetState(isExpanded ? 'collapsed' : 'half')}
        >
          <div className="w-8 h-8 rounded-full bg-blue-50 flex items-center justify-center text-blue-500 font-bold shrink-0">
            C
          </div>
          {isExpanded && (
            <span className="ml-3 font-semibold text-slate-700 whitespace-nowrap">
              资产陈列柜
            </span>
          )}
        </div>

        {/* 展柜内容区：上下分层 */}
        <div className="flex-1 overflow-y-auto scrollbar-hide py-4 px-3 space-y-6">
          {isExpanded ? (
            shelves.map((shelf) => (
              <div key={shelf.id} className="space-y-3">
                <div className="text-xs font-bold text-slate-400 tracking-wider uppercase px-2">
                  {shelf.title}
                </div>
                
                {shelf.assetIds.length === 0 ? (
                  <div className="px-2 py-4 border border-dashed border-slate-300/50 rounded-xl text-center text-xs text-slate-400 bg-white/20">
                    暂无资产
                  </div>
                ) : (
                  <div className="grid gap-2">
                    {shelf.assetIds.map((id) => {
                      const asset = assets[id];
                      if (!asset) return null;
                      return (
                        <div key={asset.id} className="glass-button p-3 rounded-xl cursor-pointer group relative overflow-hidden">
                          {/* 小魔方底座光效暗示 */}
                          <div className="absolute -bottom-4 -right-4 w-12 h-12 bg-blue-400/20 rounded-full blur-xl group-hover:bg-blue-400/40 transition-all"></div>
                          
                          <div className="text-sm font-semibold text-slate-700 truncate relative z-10">
                            {asset.title}
                          </div>
                          <div className="text-xs text-slate-500 truncate mt-1 relative z-10">
                            {asset.shortSummary}
                          </div>
                          <div className="flex gap-1 mt-2 relative z-10">
                            {asset.scenarioTags.slice(0, 2).map(tag => (
                              <span key={tag} className="text-[10px] px-1.5 py-0.5 rounded bg-slate-100 text-slate-500">
                                {tag}
                              </span>
                            ))}
                          </div>
                        </div>
                      );
                    })}
                  </div>
                )}
              </div>
            ))
          ) : (
            // 收起态的侧边图标展示
            <div className="flex flex-col items-center gap-4 pt-2">
              <div className="w-10 h-10 rounded-xl bg-white/50 border border-white flex items-center justify-center text-slate-400 hover:text-blue-500 hover:bg-white cursor-pointer transition-colors shadow-sm">
                柜
              </div>
              <div className="w-10 h-10 rounded-xl bg-white/50 border border-white flex items-center justify-center text-slate-400 hover:text-blue-500 hover:bg-white cursor-pointer transition-colors shadow-sm">
                史
              </div>
            </div>
          )}
        </div>
      </div>
    </motion.div>
  );
};
```
