### 1. 扩展 Store 以支持动画事件

#### `src/store/useCubeStore.ts` (更新)
加入魔方拧层事件 (`twistEvent`) 和叙事状态队列。

```typescript
import { create } from "zustand";
import { UiCubeModel, CubeFaceId } from "../types/cube";
import { CubeAnimationPhase } from "../types/api";

type CubeInteractionState = "idle" | "dragging" | "animating";

export interface TwistEvent {
  id: string;
  axis: 'x' | 'y' | 'z';
  layer: number; // -1, 0, 1
  angle: number; // 弧度，例如 Math.PI / 2
  duration?: number;
}

interface CubeStore {
  currentCubeView: UiCubeModel | null;
  selectedFaceId: CubeFaceId | null;
  cubeInteractionState: CubeInteractionState;
  isAnimating: boolean;
  narrativeStage: string;
  processEvents: string[];
  twistEvent: TwistEvent | null; // 新增层旋转事件

  setCurrentCubeView: (cube: UiCubeModel | null) => void;
  updateCubePhase: (phase: CubeAnimationPhase) => void;
  setSelectedFaceId: (faceId: CubeFaceId | null) => void;
  setInteractionState: (state: CubeInteractionState) => void;
  setIsAnimating: (isAnimating: boolean) => void;
  setNarrativeStage: (stage: string) => void;
  addProcessEvent: (event: string) => void;
  triggerTwist: (event: Omit<TwistEvent, 'id'>) => Promise<void>; // 触发拧层
  clearTwistEvent: () => void;
  resetCubeStore: () => void;
}

export const useCubeStore = create<CubeStore>((set) => ({
  currentCubeView: null,
  selectedFaceId: null,
  cubeInteractionState: "idle",
  isAnimating: false,
  narrativeStage: "idle",
  processEvents: [],
  twistEvent: null,

  setCurrentCubeView: (cube) => set({ currentCubeView: cube }),
  updateCubePhase: (phase) => 
    set((state) => ({
      currentCubeView: state.currentCubeView ? { ...state.currentCubeView, phase } : null
    })),
  setSelectedFaceId: (faceId) => set({ selectedFaceId: faceId }),
  setInteractionState: (state) => set({ cubeInteractionState: state }),
  setIsAnimating: (isAnimating) => set({ isAnimating }),
  setNarrativeStage: (stage) => set({ narrativeStage: stage }),
  addProcessEvent: (event) => set((state) => ({ processEvents: [...state.processEvents, event] })),
  
  // 触发层旋转并返回 Promise 方便序列编排
  triggerTwist: (event) => {
    return new Promise((resolve) => {
      set({ twistEvent: { ...event, id: Math.random().toString(36).substring(7) } });
      setTimeout(() => {
        resolve();
      }, event.duration || 600);
    });
  },
  clearTwistEvent: () => set({ twistEvent: null }),

  resetCubeStore: () => set({
    currentCubeView: null,
    selectedFaceId: null,
    cubeInteractionState: "idle",
    isAnimating: false,
    narrativeStage: "idle",
    processEvents: [],
    twistEvent: null
  })
}));
```

### 2. 动画序列编排库

#### `src/lib/animationSequences.ts`
严格按照规格文档的时序与阶段定义，编写 `8~12` 秒的异步电影级流程编排。

```typescript
import { useCubeStore } from '../store/useCubeStore';
import { useCabinetStore } from '../store/useCabinetStore';
import { CubeAnimationPhase } from '../types/api';

const delay = (ms: number) => new Promise(res => setTimeout(res, ms));

export const runOneClickDemoSequence = async () => {
  const cubeStore = useCubeStore.getState();
  const cabinetStore = useCabinetStore.getState();

  // 1. 初始化，锁定交互
  cubeStore.setIsAnimating(true);
  cubeStore.setSelectedFaceId(null);
  cubeStore.setInteractionState('animating');
  
  // ============ 阶段 1: 来源识别 ============
  cubeStore.setNarrativeStage('source_resolution');
  cubeStore.updateCubePhase('source_resolution');
  await delay(800);
  
  // 解析中打乱：U层90° -> R层-90° -> F层90°
  await cubeStore.triggerTwist({ axis: 'y', layer: 1, angle: Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'x', layer: 1, angle: -Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'z', layer: 1, angle: Math.PI / 2, duration: 500 });

  // ============ 阶段 2: 内容拆解 ============
  cubeStore.setNarrativeStage('content_reconstruction');
  cubeStore.updateCubePhase('content_reconstruction');
  await delay(1000);

  // ============ 阶段 3: 资产增强 (联动陈列柜) ============
  cubeStore.setNarrativeStage('matrix_linking');
  cubeStore.updateCubePhase('matrix_linking');
  
  // 陈列柜自动半展开亮起，模拟历史资产浮出
  cabinetStore.setCabinetState('half');
  await delay(1200);

  // 资产增强层旋转：M层90° -> U层-90° -> R层90°
  await cubeStore.triggerTwist({ axis: 'x', layer: 0, angle: Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'y', layer: 1, angle: -Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'x', layer: 1, angle: Math.PI / 2, duration: 500 });
  
  await delay(800);

  // ============ 阶段 4: 魔方成型 ============
  cubeStore.setNarrativeStage('final_form');
  cubeStore.updateCubePhase('final_form');
  
  // 最终还原旋转：R层-90° -> U层90° -> F层-90°
  await cubeStore.triggerTwist({ axis: 'x', layer: 1, angle: -Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'y', layer: 1, angle: Math.PI / 2, duration: 500 });
  await cubeStore.triggerTwist({ axis: 'z', layer: 1, angle: -Math.PI / 2, duration: 500 });

  // 退回陈列柜
  cabinetStore.setCabinetState('collapsed');

  // ============ 结束 ============
  await delay(500);
  cubeStore.setNarrativeStage('idle');
  cubeStore.setIsAnimating(false);
  cubeStore.setInteractionState('idle');
  
  // 最终视角的调整在 MatrixCube 本身监听 null 面时回正
};
```

### 3. 重写 3D 魔方组件实现真实层旋转

#### `src/components/cube/MatrixCube.tsx` (更新)
利用 Three.js `group.attach()` 动态移交子对象的所有权，完美实现数学层面和渲染层面的真实分层旋转。

```tsx
import React, { useRef, useMemo, useEffect } from 'react';
import { useFrame } from '@react-three/fiber';
import { RoundedBox } from '@react-three/drei';
import { motion } from 'framer-motion-3d';
import * as THREE from 'three';
import { useCubeStore } from '../../store/useCubeStore';
import { FACE_ROTATIONS, DEFAULT_VIEW_ANGLE, NORMAL_TO_FACE } from './constants';

const CUBIE_SIZE = 0.96;
const CUBIE_OFFSET = 1.0;

export const MatrixCube: React.FC = () => {
  const { selectedFaceId, setSelectedFaceId, twistEvent, clearTwistEvent } = useCubeStore();
  
  const mainGroupRef = useRef<THREE.Group>(null);
  const pivotGroupRef = useRef<THREE.Group>(null);
  
  // 存储所有的 cubie 引用
  const cubiesRef = useRef<THREE.Mesh[]>([]);
  // 动画状态追踪
  const animState = useRef({ isTwisting: false, targetAngle: 0, currentAngle: 0, axis: 'y' as 'x'|'y'|'z' });

  // 生成 27 个小方块初始坐标
  const initialPositions = useMemo(() => {
    const coords = [];
    for (let x = -1; x <= 1; x++) {
      for (let y = -1; y <= 1; y++) {
        for (let z = -1; z <= 1; z++) {
          coords.push(new THREE.Vector3(x * CUBIE_OFFSET, y * CUBIE_OFFSET, z * CUBIE_OFFSET));
        }
      }
    }
    return coords;
  }, []);

  const glassMaterial = useMemo(() => new THREE.MeshPhysicalMaterial({
    color: '#e0f2fe', metalness: 0.1, roughness: 0.1, transmission: 0.8, thickness: 1.5,
    clearcoat: 1.0, clearcoatRoughness: 0.1, transparent: true, opacity: 1,
  }), []);

  const coreMaterial = useMemo(() => new THREE.MeshStandardMaterial({
    color: '#38bdf8', emissive: '#0ea5e9', emissiveIntensity: 0.6,
  }), []);

  // 监听扭动事件
  useEffect(() => {
    if (twistEvent && mainGroupRef.current && pivotGroupRef.current) {
      const { axis, layer, angle, duration } = twistEvent;
      
      // 1. 重置 pivot
      pivotGroupRef.current.rotation.set(0, 0, 0);
      
      // 2. 找到属于该层的所有 cubie
      // 我们通过检测它们当前的 world position 来判断属于哪一层
      const activeCubies: THREE.Object3D[] = [];
      const threshold = 0.5; // 容差
      
      cubiesRef.current.forEach(cubie => {
        if (!cubie) return;
        const pos = new THREE.Vector3();
        cubie.getWorldPosition(pos);
        mainGroupRef.current?.worldToLocal(pos); // 转为魔方本地坐标系
        
        const posValue = pos[axis];
        if (Math.abs(posValue - (layer * CUBIE_OFFSET)) < threshold) {
          activeCubies.push(cubie);
        }
      });

      // 3. 将选中的 cubies attach 到 pivot 上（保留世界坐标视觉位置）
      activeCubies.forEach(cubie => {
        pivotGroupRef.current?.attach(cubie);
      });

      // 4. 开启每帧动画
      animState.current = {
        isTwisting: true,
        targetAngle: angle,
        currentAngle: 0,
        axis
      };

      // 5. 动画结束清理
      setTimeout(() => {
        animState.current.isTwisting = false;
        // 修正精确角度
        pivotGroupRef.current!.rotation[axis] = angle;
        // 把 cubie 重新 attach 回主组
        activeCubies.forEach(cubie => {
          mainGroupRef.current?.attach(cubie);
        });
        // 回收事件
        clearTwistEvent();
      }, duration || 600);
    }
  }, [twistEvent, clearTwistEvent]);

  // 每帧处理层旋转动画和呼吸动画
  useFrame((state, delta) => {
    // 拧层插值动画
    if (animState.current.isTwisting && pivotGroupRef.current) {
      const { targetAngle, axis } = animState.current;
      // 简单平滑插值 (Lerp)
      pivotGroupRef.current.rotation[axis] = THREE.MathUtils.lerp(
        pivotGroupRef.current.rotation[axis], 
        targetAngle, 
        delta * 10 // 调整速度
      );
    }

    // 待机呼吸微动
    if (!selectedFaceId && mainGroupRef.current && !twistEvent) {
      mainGroupRef.current.position.y = Math.sin(state.clock.elapsedTime * 1.5) * 0.05;
    } else if (mainGroupRef.current) {
      mainGroupRef.current.position.y = THREE.MathUtils.lerp(mainGroupRef.current.position.y, 0, 0.1);
    }
  });

  const handleCubieClick = (e: any) => {
    e.stopPropagation();
    if (twistEvent) return; // 动画中屏蔽点击
    if (e.face) {
      const normal = e.face.normal.clone().round();
      const faceId = NORMAL_TO_FACE[`${normal.x},${normal.y},${normal.z}`];
      if (faceId) setSelectedFaceId(faceId);
    }
  };

  const targetRotation = selectedFaceId ? FACE_ROTATIONS[selectedFaceId] : DEFAULT_VIEW_ANGLE;

  return (
    <motion.group
      ref={mainGroupRef}
      animate={{ rotateX: targetRotation[0], rotateY: targetRotation[1], rotateZ: targetRotation[2] }}
      transition={{ type: 'spring', stiffness: 60, damping: 15 }}
    >
      {/* 隐藏的 Pivot 组，仅用于挂载正在旋转的层 */}
      <group ref={pivotGroupRef} />

      {initialPositions.map((pos, idx) => {
        const isCore = pos.x === 0 && pos.y === 0 && pos.z === 0;
        return (
          <group key={idx} position={pos}>
            <RoundedBox
              ref={(el) => (cubiesRef.current[idx] = el as any)}
              args={[CUBIE_SIZE, CUBIE_SIZE, CUBIE_SIZE]}
              radius={0.05}
              smoothness={4}
              onClick={handleCubieClick}
              material={isCore ? coreMaterial : glassMaterial}
            />
          </group>
        );
      })}
    </motion.group>
  );
};
```

### 4. 接入到底部输入条

#### `src/components/layout/BottomInputBar.tsx` (更新)
把我们写好的 `runOneClickDemoSequence` 绑到“一键体验”逻辑上，并处理禁用状态。

```tsx
import React, { useState } from 'react';
import { runOneClickDemoSequence } from '../../lib/animationSequences';
import { useCubeStore } from '../../store/useCubeStore';

export const BottomInputBar: React.FC = () => {
  const [inputValue, setInputValue] = useState('');
  const { isAnimating } = useCubeStore();

  const handleReconstruct = async () => {
    if (isAnimating) return;
    
    if (inputValue.trim() === 'demo') {
      // 触发规格要求的"一键体验"全链路流程
      await runOneClickDemoSequence();
    } else {
      // 普通的轻量重构 / fallback
      await runOneClickDemoSequence(); // MVP暂用同一个做视觉展出
    }
  };

  return (
    <div className="absolute bottom-8 left-1/2 -translate-x-1/2 z-40 w-full max-w-2xl">
      <div className="glass-panel rounded-2xl p-2 flex items-center gap-2 shadow-lg shadow-blue-900/5">
        <div className="flex-1 px-4 py-2">
          <input
            type="text"
            value={inputValue}
            onChange={(e) => setInputValue(e.target.value)}
            disabled={isAnimating}
            placeholder="粘贴抖音链接 / 输入目标 / 拖入本地视频..."
            className="w-full bg-transparent outline-none text-slate-700 placeholder:text-slate-400 disabled:opacity-50"
            onKeyDown={(e) => e.key === 'Enter' && handleReconstruct()}
          />
        </div>
        <div className="w-px h-6 bg-slate-300/50 mx-2"></div>
        
        {/* 提供弱化的 一键体验 快捷按钮 */}
        <button 
          onClick={() => { setInputValue('demo'); runOneClickDemoSequence(); }}
          disabled={isAnimating}
          className="text-blue-600 text-sm font-medium px-4 hover:bg-blue-50 py-2 rounded-xl transition-colors disabled:opacity-50"
        >
          一键体验
        </button>

        <button 
          onClick={handleReconstruct}
          disabled={isAnimating}
          className="bg-blue-600 hover:bg-blue-500 disabled:bg-slate-400 text-white px-6 py-2 rounded-xl text-sm font-medium transition-colors shadow-sm shadow-blue-500/30"
        >
          {isAnimating ? '正在重构...' : '开始重构'}
        </button>
      </div>
    </div>
  );
};
```
