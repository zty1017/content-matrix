import { CubeFaceId } from '../../types/api';

// 各个面的默认旋转角度 (Euler: [x, y, z])
export const FACE_ROTATIONS: Record<CubeFaceId, [number, number, number]> = {
  primary_card: [0, 0, 0],                // 前面 (+Z)
  snapshot: [0, Math.PI, 0],              // 后面 (-Z)

  evidence: [0, -Math.PI / 2, 0],         // 右面 (+X)
  source: [Math.PI / 2, 0, 0],            // 顶面 (+Y)
  inferences: [-Math.PI / 2, 0, 0]        // 底面 (-Y)
};

// 默认 3/4 空间工作台视角
export const DEFAULT_VIEW_ANGLE: [number, number, number] = [Math.PI / 6, -Math.PI / 4, 0];

// 面法向量到面 ID 的映射（用于射线检测）
export const NORMAL_TO_FACE: Record<string, CubeFaceId> = {
  "0,0,1": "primary_card",
  "0,0,-1": "snapshot",
  "-1,0,0": "related_assets",
  "1,0,0": "evidence",
  "0,1,0": "source",
  "0,-1,0": "inferences"
};

### 2. 3D 魔方实现逻辑

#### `src/components/cube/MatrixCube.tsx`
魔方本体。包含 27 个小方块、琉璃材质、以及利用 `framer-motion-3d` 控制的整体旋转和用于系统动画的层旋转基建。
*(注意：我们需要安装 `framer-motion-3d`，它是 Framer Motion 官方提供的 3D 扩展。如果环境中只有 `framer-motion`，一般在 v10+ 中自带 3D 功能。)*

```tsx
import React, { useRef, useMemo, useEffect } from 'react';
import { useFrame } from '@react-three/fiber';
import { RoundedBox } from '@react-three/drei';
import { motion } from 'framer-motion-3d';
import * as THREE from 'three';
import { useCubeStore } from '../../store/useCubeStore';
import { FACE_ROTATIONS, DEFAULT_VIEW_ANGLE, NORMAL_TO_FACE } from './constants';

const CUBIE_SIZE = 0.96; // 留出 0.04 的清晰缝隙
const CUBIE_OFFSET = 1.0;

export const MatrixCube: React.FC = () => {
  const { selectedFaceId, setSelectedFaceId } = useCubeStore();
  const groupRef = useRef<THREE.Group>(null);

  // 生成 3x3x3 = 27 个小方块的初始坐标
  const cubies = useMemo(() => {
    const coords = [];
    for (let x = -1; x <= 1; x++) {
      for (let y = -1; y <= 1; y++) {
        for (let z = -1; z <= 1; z++) {
          coords.push(new THREE.Vector3(x * CUBIE_OFFSET, y * CUBIE_OFFSET, z * CUBIE_OFFSET));
        }
      }
    }
    return coords;
  }, []);

  // 琉璃质感材质配置 (MeshPhysicalMaterial)
  const glassMaterial = useMemo(() => new THREE.MeshPhysicalMaterial({
    color: '#e0f2fe',          // 浅冷蓝白
    metalness: 0.1,
    roughness: 0.1,            // 光滑表面
    transmission: 0.8,         // 玻璃透光
    thickness: 1.5,            // 折射厚度
    clearcoat: 1.0,            // 表面清漆高光
    clearcoatRoughness: 0.1,
    transparent: true,
    opacity: 1,
  }), []);

  // 内部发光骨架材质
  const coreMaterial = useMemo(() => new THREE.MeshStandardMaterial({
    color: '#38bdf8',
    emissive: '#0ea5e9',
    emissiveIntensity: 0.5,
  }), []);

  // 点击外表面的处理逻辑
  const handleCubieClick = (e: any) => {
    e.stopPropagation();
    // 获取点击的面的法向量 (world normal 转换到 local normal)
    if (e.face) {
      const normal = e.face.normal.clone().round();
      const normalKey = `${normal.x},${normal.y},${normal.z}`;
      const faceId = NORMAL_TO_FACE[normalKey];
      if (faceId) {
        setSelectedFaceId(faceId);
      }
    }
  };

  // 根据选中的面计算目标旋转角度
  const targetRotation = selectedFaceId ? FACE_ROTATIONS[selectedFaceId] : DEFAULT_VIEW_ANGLE;

  // 待机呼吸微动（极小幅度上下悬浮）
  useFrame((state) => {
    if (!selectedFaceId && groupRef.current) {
      groupRef.current.position.y = Math.sin(state.clock.elapsedTime * 1.5) * 0.05;
    } else if (groupRef.current) {
      groupRef.current.position.y = THREE.MathUtils.lerp(groupRef.current.position.y, 0, 0.1);
    }
  });

  return (
    <motion.group
      ref={groupRef}
      // 使用 framer-motion-3d 进行平滑的面追踪旋转
      animate={{
        rotateX: targetRotation[0],
        rotateY: targetRotation[1],
        rotateZ: targetRotation[2],
      }}
      transition={{ type: 'spring', stiffness: 60, damping: 15 }}
      // 当点击魔方外部的空白区域时，需要取消选中并回正 (在 Scene 层处理)
    >
      {cubies.map((pos, idx) => {
        const isCore = pos.x === 0 && pos.y === 0 && pos.z === 0;
        return (
          <group key={idx} position={pos}>
            <RoundedBox
              args={[CUBIE_SIZE, CUBIE_SIZE, CUBIE_SIZE]}
              radius={0.05}
              smoothness={4}
              onClick={handleCubieClick}
              material={isCore ? coreMaterial : glassMaterial}
            >
              {/* 这里可以通过 Edges 来增强科技感边缘线，但为了极简，我们先靠高光折射 */}
            </RoundedBox>
          </group>
        );
      })}
    </motion.group>
  );
};
```

### 3. 3D 场景画布包裹器

#### `src/components/cube/CubeScene.tsx`
初始化 R3F 环境，加入环境光、控制轨道（支持拖拽但不乱翻），并接管空白区域点击。

```tsx
import React, { Suspense } from 'react';
import { Canvas } from '@react-three/fiber';
import { OrbitControls, Environment, ContactShadows } from '@react-three/drei';
import { MatrixCube } from './MatrixCube';
import { useCubeStore } from '../../store/useCubeStore';

export const CubeScene: React.FC = () => {
  const { setSelectedFaceId } = useCubeStore();

  // 点击空白区域，还原到 3/4 视角
  const handlePointerMissed = () => {
    setSelectedFaceId(null);
  };

  return (
    <div className="absolute inset-0 z-10">
      <Canvas 
        camera={{ position: [0, 0, 8], fov: 45 }}
        onPointerMissed={handlePointerMissed}
        gl={{ antialias: true, alpha: true }}
      >
        <Suspense fallback={null}>
          {/* 环境光与直射光，营造浅冷调 */}
          <ambientLight intensity={0.6} color="#ffffff" />
          <directionalLight position={[10, 10, 5]} intensity={1.5} color="#e0f2fe" />
          <directionalLight position={[-10, -10, -5]} intensity={0.5} color="#bae6fd" />
          
          {/* 环境贴图，对于玻璃/琉璃材质极其重要 */}
          <Environment preset="city" />

          {/* 主魔方 */}
          <MatrixCube />

          {/* 底部柔和接触阴影 */}
          <ContactShadows 
            position={[0, -2.5, 0]} 
            opacity={0.3} 
            scale={10} 
            blur={2.5} 
            far={4} 
            color="#0f172a"
          />

          {/* 轨道控制器：允许整体拖拽旋转。当选中具体面时，稍微限制它以避免打断对焦 */}
          <OrbitControls 
            enablePan={false}
            enableZoom={false}
            minPolarAngle={Math.PI / 4}
            maxPolarAngle={Math.PI - Math.PI / 4}
            makeDefault
          />
        </Suspense>
      </Canvas>
    </div>
  );
};
```

### 当前实现亮点验证
1. **真实三阶结构**：通过 27 个 `RoundedBox` 生成，而不是贴图，符合 `3×3×3 结构清晰，保留缝隙` 的要求。
2. **材质视觉**：使用 `MeshPhysicalMaterial` 开启 `transmission`，配上 Drei 的 `Environment`，无需复杂配置即可得到通透高亮的**冷白冰蓝琉璃质感**。核心小块发微光。
3. **点击面吸附**：计算了法向量 (`NORMAL_TO_FACE`) 并绑定了 `onClick`。点击任意表面小方块，会触发状态流更新 `selectedFaceId`，接着 `framer-motion-3d` 的 `<motion.group>` 会弹性旋转到该面的正视角。
4. **待机动画**：未点击特定面时（`selectedFaceId == null`），除了 OrbitControls 可以自由拖拽外，魔方还在极小幅度地悬浮呼吸（`useFrame` 里的正弦波逻辑）。

*(注：系统打乱/还原的真实层旋转动画逻辑稍微复杂，属于动画序列 sequence 统筹部分。这套组件中 27 个独立 cubies 已经为其做好了分组和 Pivot 旋转的底层数据准备，我们将在接入"一键体验"时完成具体的打乱编排。)*
